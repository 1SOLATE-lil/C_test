#include "raylib.h"

#define SCREEN_WIDTH 800
#define SCREEN_HEIGHT 600

#define BLOCK_ROWS 6
#define BLOCK_COLS 8

#define BLOCK_WIDTH 80
#define BLOCK_HEIGHT 25
#define BLOCK_GAP 10

#define BALL_RADIUS 8

typedef struct Block {
    Rectangle rect;
    int active;
    Color color;
} Block;

void ResetBall(Vector2* ball, Vector2* ballSpeed) {
    ball->x = SCREEN_WIDTH / 2;
    ball->y = 420;

    ballSpeed->x = 4;
    ballSpeed->y = -4;
}

int main(void) {
    InitWindow(SCREEN_WIDTH, SCREEN_HEIGHT, "Rainbow Block Breaker");
    SetTargetFPS(60);

    Rectangle paddle = { 340, 540, 120, 18 };

    Vector2 ball;
    Vector2 ballSpeed;
    ResetBall(&ball, &ballSpeed);

    Block blocks[BLOCK_ROWS][BLOCK_COLS];

    Color rainbow[BLOCK_ROWS] = {
        RED,
        ORANGE,
        YELLOW,
        GREEN,
        BLUE,
        PURPLE
    };

    int startX = 45;
    int startY = 80;

    for (int i = 0; i < BLOCK_ROWS; i++) {
        for (int j = 0; j < BLOCK_COLS; j++) {
            blocks[i][j].rect.x = startX + j * (BLOCK_WIDTH + BLOCK_GAP);
            blocks[i][j].rect.y = startY + i * (BLOCK_HEIGHT + BLOCK_GAP);
            blocks[i][j].rect.width = BLOCK_WIDTH;
            blocks[i][j].rect.height = BLOCK_HEIGHT;
            blocks[i][j].active = 1;
            blocks[i][j].color = rainbow[i];
        }
    }

    int score = 0;
    int life = 5;
    int gameOver = 0;
    int gameClear = 0;

    while (!WindowShouldClose()) {
        if (!gameOver && !gameClear) {
            if (IsKeyDown(KEY_LEFT) || IsKeyDown(KEY_A)) {
                paddle.x -= 7;
            }

            if (IsKeyDown(KEY_RIGHT) || IsKeyDown(KEY_D)) {
                paddle.x += 7;
            }

            if (paddle.x < 0) {
                paddle.x = 0;
            }

            if (paddle.x + paddle.width > SCREEN_WIDTH) {
                paddle.x = SCREEN_WIDTH - paddle.width;
            }

            ball.x += ballSpeed.x;
            ball.y += ballSpeed.y;

            if (ball.x <= BALL_RADIUS || ball.x >= SCREEN_WIDTH - BALL_RADIUS) {
                ballSpeed.x *= -1;
            }

            if (ball.y <= BALL_RADIUS) {
                ballSpeed.y *= -1;
            }

            if (CheckCollisionCircleRec(ball, BALL_RADIUS, paddle) && ballSpeed.y > 0) {
                ballSpeed.y *= -1;

                float paddleCenter = paddle.x + paddle.width / 2;
                float distance = ball.x - paddleCenter;

                ballSpeed.x = distance / 15;
            }

            int remainBlocks = 0;

            for (int i = 0; i < BLOCK_ROWS; i++) {
                for (int j = 0; j < BLOCK_COLS; j++) {
                    if (blocks[i][j].active == 1) {
                        remainBlocks++;

                        if (CheckCollisionCircleRec(ball, BALL_RADIUS, blocks[i][j].rect)) {
                            blocks[i][j].active = 0;
                            score += 10;
                            ballSpeed.y *= -1;
                        }
                    }
                }
            }

            if (remainBlocks == 0) {
                gameClear = 1;
            }

            if (ball.y > SCREEN_HEIGHT + BALL_RADIUS) {
                life--;

                if (life <= 0) {
                    gameOver = 1;
                }
                else {
                    ResetBall(&ball, &ballSpeed);
                    paddle.x = 340;
                }
            }
        }

        BeginDrawing();

        ClearBackground(BLACK);

        DrawText("Rainbow Block Breaker", 250, 20, 25, WHITE);

        DrawText(TextFormat("Score: %d", score), 20, 20, 25, WHITE);
        DrawText(TextFormat("Life: %d", life), 680, 20, 25, WHITE);

        for (int i = 0; i < BLOCK_ROWS; i++) {
            for (int j = 0; j < BLOCK_COLS; j++) {
                if (blocks[i][j].active == 1) {
                    DrawRectangleRec(blocks[i][j].rect, blocks[i][j].color);
                    DrawRectangleLinesEx(blocks[i][j].rect, 2, WHITE);
                }
            }
        }

        DrawRectangleRec(paddle, WHITE);
        DrawCircleV(ball, BALL_RADIUS, WHITE);

        if (gameOver) {
            DrawText("GAME OVER", 300, 260, 45, RED);
            DrawText(TextFormat("Final Score: %d", score), 300, 320, 25, WHITE);
            DrawText("Press ESC to exit", 310, 360, 20, GRAY);
        }

        if (gameClear) {
            DrawText("GAME CLEAR!", 280, 260, 45, GREEN);
            DrawText(TextFormat("Final Score: %d", score), 300, 320, 25, WHITE);
            DrawText("Press ESC to exit", 310, 360, 20, GRAY);
        }

        EndDrawing();
    }

    CloseWindow();

    return 0;
}

이 게임은 C언어와 Raylib 라이브러리를 사용하여 제작한 무지개 블록깨기 게임이다.
플레이어는 키보드 방향키로 패들을 움직여 공을 튕기고, 화면 위쪽의 무지개색 블록을 제거한다.
블록을 제거할 때마다 점수가 10점씩 증가하며, 현재 점수와 남은 목숨은 화면 상단에 출력된다.
공을 놓치면 목숨이 1 감소하고, 목숨이 모두 사라지면 게임이 종료된다.
모든 블록을 제거하면 게임 클리어 화면이 출력된다.

조작법
← 또는 A : 왼쪽 이동
→ 또는 D : 오른쪽 이동
ESC : 종료
