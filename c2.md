```
#include <stdio.h>

struct Student {
    char name[20];
    int kor;
    int eng;
    int math;
};

int main() {
    struct Student s[2] = {
        {"Kim", 80, 90, 70},
        {"Lee", 90, 85, 95}
    };

    double korAvg, engAvg, mathAvg;
    double studentAvg1, studentAvg2;

    korAvg = (s[0].kor + s[1].kor) / 2.0;
    engAvg = (s[0].eng + s[1].eng) / 2.0;
    mathAvg = (s[0].math + s[1].math) / 2.0;

    studentAvg1 = (s[0].kor + s[0].eng + s[0].math) / 3.0;
    studentAvg2 = (s[1].kor + s[1].eng + s[1].math) / 3.0;

    printf("=== 학생 성적 ===\n");

    printf("%s: 국어 %d, 영어 %d, 수학 %d, 평균 %.2f\n",
        s[0].name, s[0].kor, s[0].eng, s[0].math, studentAvg1);

    printf("%s: 국어 %d, 영어 %d, 수학 %d, 평균 %.2f\n",
        s[1].name, s[1].kor, s[1].eng, s[1].math, studentAvg2);

    printf("\n=== 과목 평균 ===\n");
    printf("국어 평균: %.2f\n", korAvg);
    printf("영어 평균: %.2f\n", engAvg);
    printf("수학 평균: %.2f\n", mathAvg);

    return 0;
}
```

이 프로그램은 구조체를 활용하여 학생 2명의 이름과 국어, 영어, 수학 성적을 저장하는 프로그램이다.
Student 구조체에는 학생의 이름과 세 과목 점수를 저장할 수 있도록 name, kor, eng, math 변수를 선언하였다.
이후 구조체 배열 s[2]를 사용하여 두 학생의 정보를 초기화하였다.

프로그램에서는 각 학생의 세 과목 점수를 더한 뒤 3으로 나누어 학생별 평균을 계산하고, 같은 과목끼리 점수를 더한 뒤 학생 수 2로 나누어 과목별 평균을 계산한다.
마지막으로 printf() 함수를 사용하여 학생별 성적과 평균, 그리고 국어, 영어, 수학의 과목 평균을 화면에 출력한다.
